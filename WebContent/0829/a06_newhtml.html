이벤트 핸들러 작성(등록) 방법
	1) bind : body 태그 안에 실행 전부터 이미 만들어져 있는 요소
		$('p').bind('click', function(){})
		$('p').on('click', function(){})
		$('p').click(function(){})
	2) delegate : 실행 후 새롭게 동적으로 만들어지는 요소
		$('부모태그').on('click', '대상태그', function(){}) 
			ex) $('body').delegate('p', 'click', function(){})
		$('부모태그').delegate('대상태그', 'click', function(){})
			ex) $('body').on('click', 'p', function(){})
